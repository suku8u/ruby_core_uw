def xml_result (row)
  http = HTTPClient.new
  @url = "#{@url_static}origin=#{Distance.origin_latitude (row)},#{Distance.origin_longitude (row)}\
          &destination=#{Distance.origin_latitude (row)},#{Distance.destination_longitude (row)}&sensor=false"
  @xml_result = open(@url.delete(' ')).read
  return @xml_result
end
